fmt

Go格式化输出符

1 Verb

verb explanation
%d int 类型十进制输出
%o, %b int 类型的八进制、二进制输出
%x, %X 小/大写的十六进制表示
%p 加上前导 0x 的十六进制，常用于指针地址表示
%f 浮点数的单精度
%b 浮点数的二进制科学计数法
%e, %E 科学计数法小写、大写
%g, %G 根据情况选择 %e、%E 或 %f 以产生更紧凑的（无末尾的0）输出
%t 布尔类型输出，true or false
%c rune 类型输出，即对应的 unicode 码值
%U 输出 unicode 格式
%s 字符串
%q 带双引号的字符串或单引号的 rune
%v 将任意类型输出默认格式
%+v 在输出结构体时添加字段名
%#v 对应的 Go 语法表示
%T 输出类型
%% 输出 % 自身
%c 输入中的占位符

2 Symbol

symbol explanation
+ 打印数值的正负号
- 打印内容左对其
# 为八进制添加前导0，十六进制添加前导0x，为 %p 去掉前导 0x，对于 %q 在转换成功情况下打印原始字符串，%#U 输出 unicode 编码形式
0 填充前导 0

3 fmt

3.1 Input

function explanation
func Scan(a ...interface{}) (n int, err error) 从标准输入读取，将数据空格分割后存入 a，EOF 停止
func Scanln(a ...interface{}) (n int, err error) 从标准输入读取，将数据空格分割后存入 a，\n 停止
func Scanf(format string, a ...interface{}) (n int, err error) 从标准输入读取，根据 format 对数据进行解析，存入 a
func Fscan(r io.Reader, a ...interface{}) (n int, err error) 同 Scan，从 io.Reader 中读取数据
func Fscanln(r io.Reader, a ...interface{}) (n int, err error) 同 Scanln，从 io.Reader 中读取数据
func Fscanf(r io.Reader, format string, a ...interface{}) (n int, err error) 同 Scanf，从 io.Reader 中读取数据
func Sscan(str string, a ...interface{}) (n int, err error) 同 Scan，从 string 中读取数据
func Sscanln(str string, a ...interface{}) (n int, err error) 同 Scanln，从 string 中读取数据
func Sscanf(str string, format string, a ...interface{}) (n int, err error) 同 Scanf，从 string 中读取数据

3.2 Output

function explanation
func Print(a ...interface{}) (n int, err error) 输出到 os.stdout，非字符串参数之间添加空格
func Println(a ...interface{}) (n int, err error) 输出到 os.stdout，输出末尾添加换行符
func Printf(format string, a ...interface{}) (n int, err error) 输出到 os.stdout，格式化输出
func Fprint(w io.Writer, a ...interface{}) (n int, err error) 同 Print，输出到 io.Writer
func Fprintln(w io.Writer, a ...interface{}) (n int, err error) 同 Println，输出到 io.Writer
func Fprintf(w io.Writer, format string, a ...interface{}) (n int, err error) 同 Printf，输出到 io.Writer
func Sprint(a ...interface{}) string 同 Print，将结果以字符串方式返回
func Sprintln(a ...interface{}) string 同 Println，将结果以字符串方式返回
func Sprintf(format string, a ...interface{}) string 同 Printf，将结果以字符串方式返回
func Errorf(format string, a ...interface{}) error 返回 error 类型的结果字符串
